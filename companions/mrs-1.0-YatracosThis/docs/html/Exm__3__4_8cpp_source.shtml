00037 AIA_BOOL_INTERVAL IBT_ex3_4(const ivector& x)
00038 {
00039   // here we test a 2-d box for inclusion in the area 
00040   // such that x1^4 - x1^2 + 4x2^2 is in the interval [-0.1, 0.1]
00041 
00042   interval ToInvert(-0.1, 0.1),Temp;
00043   interval Img = power(x[1],4) - sqr(x[1]) + 4*sqr(x[2]);
00044 
00045   if (!Intersection(Temp,Img,ToInvert)) return BI_FALSE;
00046   if ( Img<=ToInvert ) return BI_TRUE;
00047 
00048   return BI_INDET;
00049 }

00051 AIA_BOOL_INTERVAL IBTFinverse_ex3_4(const ivector& x)
00052 {
00053   ivector Img(2);
00054   // example in 2-d space R2
00055   // for f: R2 -> R2
00056   //f1(x1, x2) = (x1-1)^2 - 1+ x2
00057   //f2(x1, x2) = -(x1^2) + (x2-1)^2
00058 
00059   Img[1] = sqr(x[1]) - 2*x[1] + x[2];
00060   Img[2] = -sqr(x[1]) + sqr(x[2]) - 2*x[2] + 1;
00061 
00062   return (Img<=Sc6);
00063 }
